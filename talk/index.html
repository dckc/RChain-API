JSON to RHOCore: recur


    function recur(x) {
        switch (typeof x) {
        case 'boolean':
            return expr1({ g_bool: x });
        case 'number':
            return expr1({ g_int: x }); // ISSUE: only integers
        case 'string':
            return expr1({ g_string: x });
        case 'object':
            if (x === null) {
                return fixLF({});
            } else if (Array.isArray(x)) {
                return toArry(x);
            } else {
                return keysValues(x);
            }
        default:
            throw(new Error('no mapping to RSON for ' +  typeof x));
        };
    }
